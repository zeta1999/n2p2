111 void Training::selectSets()
112 {
113  log << "\n";
114  log << "*** DEFINE TRAINING/TEST SETS ***********"
115  "**************************************\n";
116  log << "\n";
117 
118  size_t numMyEnergiesTrain = 0;
119  size_t numMyEnergiesTest = 0;
120  size_t numMyForcesTrain = 0;
121  size_t numMyForcesTest = 0;
122  vector<size_t> numMyForcesTrainPerElement;
123  numMyForcesTrainPerElement.resize(numElements, 0);
124  double testSetFraction = atof(settings["test_fraction"].c_str());
125  log << strpr("Desired test set ratio : %f\n", testSetFraction);
126  //for (vector<Structure>::iterator it = structures.begin();
127  // it != structures.end(); ++it)
128  if (structures.size() > 0) hasStructures = true;
129  else hasStructures = false;
130  for (size_t i = 0; i < structures.size(); ++i)
131  {
132  Structure& s = structures.at(i);
133  if (gsl_rng_uniform(rng) < testSetFraction)
134  {
135  s.sampleType = Structure::ST_TEST;
136  numMyEnergiesTest++;
137  numMyForcesTest += 3 * s.numAtoms;
138  }
139  else
140  {
141  s.sampleType = Structure::ST_TRAINING;
142  numMyEnergiesTrain++;
143  numMyForcesTrain += 3 * s.numAtoms;
144  for (size_t j = 0; j < numElements; ++j)
145  {
146  numMyForcesTrainPerElement.at(j) +=
147  3 * s.numAtomsPerElement.at(j);
148  }
149  updateCandidatesEnergy.push_back(UpdateCandidate());
150  updateCandidatesEnergy.back().s = i;
151  for (vector<Atom>::const_iterator it = s.atoms.begin();
152  it != s.atoms.end(); ++it)
153  {
154  for (size_t j = 0; j < 3; ++j)
155  {
156  updateCandidatesForce.push_back(UpdateCandidate());
157  updateCandidatesForce.back().s = i;
158  updateCandidatesForce.back().a = it->index;
159  updateCandidatesForce.back().c = j;
160  }
161  }
162  }
163  }
164  for (size_t i = 0; i < numElements; ++i)
165  {
166  if (hasStructures && (numMyForcesTrainPerElement.at(i) == 0))
167  {
168  log << strpr("WARNING: Process %d has no atoms of element "
169  "%d (%2s).\n",
170  myRank,
171  i,
172  elementMap[i].c_str());
173  }
174  }
175  MPI_Allreduce(MPI_IN_PLACE, &numMyEnergiesTrain, 1, MPI_SIZE_T, MPI_SUM, comm);
176  MPI_Allreduce(MPI_IN_PLACE, &numMyEnergiesTest , 1, MPI_SIZE_T, MPI_SUM, comm);
177  MPI_Allreduce(MPI_IN_PLACE, &numMyForcesTrain , 1, MPI_SIZE_T, MPI_SUM, comm);
178  MPI_Allreduce(MPI_IN_PLACE, &numMyForcesTest , 1, MPI_SIZE_T, MPI_SUM, comm);
179  numEnergiesTrain = numMyEnergiesTrain;
180  numForcesTrain = numMyForcesTrain;
181  log << strpr("Total number of energies : %d\n", numStructures);
182  log << strpr("Number of training energies : %d\n", numMyEnergiesTrain);
183  log << strpr("Number of test energies : %d\n", numMyEnergiesTest);
184  log << strpr("Number of training forces : %d\n", numMyForcesTrain);
185  log << strpr("Number of test forces : %d\n", numMyForcesTest);
186  log << strpr("Actual test set fraction : %f\n",
187  numMyEnergiesTest / double(numStructures));
188 
189  log << "*****************************************"
190  "**************************************\n";
191 
192  return;
193 }

195 void Training::writeSetsToFiles()

198  log << "*** WRITE TRAINING/TEST SETS ************"

200  log << "\n";
201 
202  string fileName = strpr("train.data.%04d", myRank);
203  ofstream fileTrain;
204  fileTrain.open(fileName.c_str());
205  if (!fileTrain.is_open())
206  {
207  runtime_error(strpr("ERROR: Could not open file %s\n",
208  fileName.c_str()));
209  }
210  fileName = strpr("test.data.%04d", myRank);
211  ofstream fileTest;
212  fileTest.open(fileName.c_str());
213  if (!fileTest.is_open())
214  {
215  runtime_error(strpr("ERROR: Could not open file %s\n",
216  fileName.c_str()));
217  }
218  for (vector<Structure>::iterator it = structures.begin();
219  it != structures.end(); ++it)
220  {
221  // Energy offsets are already subtracted at this point.
222  // Here, we quickly add them again to provide consistent data sets.
223  addEnergyOffset(*it);
224  if (it->sampleType == Structure::ST_TRAINING)
225  {
226  it->writeToFile(&fileTrain);
227  }
228  else if (it->sampleType == Structure::ST_TEST)
229  {
230  it->writeToFile(&fileTest);
231  }
232  // Subtract energy offsets again.
233  removeEnergyOffset(*it);
234  }
235  fileTrain.flush();
236  fileTrain.close();
237  fileTest.flush();
238  fileTest.close();
239  MPI_Barrier(comm);
240  if (myRank == 0)
241  {
242  log << "Writing training/test set to files:\n";
243  log << " - train.data\n";
244  log << " - test.data\n";
245  fileName = "train.data";
246  combineFiles(fileName);
247  fileName = "test.data";
248  combineFiles(fileName);
249  }
250 
251  log << "*****************************************"
252  "**************************************\n";
253 
254  return;

257 void Training::initializeWeights()
258 {
259  // Local weights array (actual weights array depends on update strategy).
260  vector<vector<double> > w;
261 
262  log << "\n";
263  log << "*** WEIGHT INITIALIZATION ***************"
264  "**************************************\n";
265  log << "\n";
266 
267  if (settings.keywordExists("nguyen_widrow_weights_short") &&
268  settings.keywordExists("precondition_weights"))
269  {
270  throw runtime_error("ERROR: Nguyen Widrow and preconditioning weights"
271  " initialization are incompatible\n");
272  }
273 
274  // Create per-element connections vectors.
275  for (size_t i = 0; i < numElements; ++i)
276  {
277  w.push_back(vector<double>());
278  w.at(i).resize(elements.at(i).neuralNetwork->getNumConnections(), 0.0);
279  }
280 
281  if (settings.keywordExists("use_old_weights_short"))
282  {
283  log << "Reading old weights from files.\n";
284  log << "Calling standard weight initialization routine...\n";
285  log << "*****************************************"
286  "**************************************\n";
287  setupNeuralNetworkWeights("weights.%03zu.data");
288  return;
289  }
290  else
291  {
292  double minWeights = atof(settings["weights_min"].c_str());
293  double maxWeights = atof(settings["weights_max"].c_str());
294  log << strpr("Initial weights selected randomly in interval "
295  "[%f, %f).\n", minWeights, maxWeights);
296  for (size_t i = 0; i < numElements; ++i)
297  {
298  for (size_t j = 0; j < w.at(i).size(); ++j)
299  {
300  w.at(i).at(j) = minWeights + gsl_rng_uniform(rngGlobal)
301  * (maxWeights - minWeights);
302  }
303  elements.at(i).neuralNetwork->setConnections(&(w.at(i).front()));
304  }
305  if (settings.keywordExists("nguyen_widrow_weights_short"))
306  {
307  log << "Weights modified according to Nguyen Widrow scheme.\n";
308  for (vector<Element>::iterator it = elements.begin();
309  it != elements.end(); ++it)
310  {
311  it->neuralNetwork->
312  modifyConnections(NeuralNetwork::MS_NGUYENWIDROW);
313  }
314  }
315  else if (settings.keywordExists("precondition_weights"))
316  {
317  throw runtime_error("ERROR: Preconditioning of weights not yet"
318  " implemented.\n");
319  //it->neuralNetwork->
320  // modifyConnections(NeuralNetwork::MS_PRECONDITIONOUTPUT,
321  // mean,
322  // sigma);
323  }
324  else
325  {
326  log << "Weights modified accoring to Glorot Bengio scheme.\n";
327  //log << "Weights connected to output layer node set to zero.\n";
328  log << "Biases set to zero.\n";
329  for (vector<Element>::iterator it = elements.begin();
330  it != elements.end(); ++it)
331  {
332  it->neuralNetwork->
333  modifyConnections(NeuralNetwork::MS_GLOROTBENGIO);
334  //it->neuralNetwork->
335  // modifyConnections(NeuralNetwork::MS_ZEROOUTPUTWEIGHTS);
336  it->neuralNetwork->
337  modifyConnections(NeuralNetwork::MS_ZEROBIAS);
338  }
339  }
340  }
341 
342  log << "*****************************************"
343  "**************************************\n";
344 
345  return;
346 }

393 void Training::setupTraining()
394 {
395  log << "\n";
396  log << "*** SETUP: TRAINING *********************"
397  "**************************************\n";
398  log << "\n";
399 
400  useForces = settings.keywordExists("use_short_forces");
401  if (useForces)
402  {
403  log << "Forces will be used for training.\n";
404  if (settings.keywordExists("force_weight"))
405  {
406  forceWeight = atof(settings["force_weight"].c_str());
407  }
408  else
409  {
410  log << "WARNING: Force weight not set, using default value.\n";
411  forceWeight = 1.0;
412  }
413  log << strpr("Force update weight: %10.2E\n", forceWeight);
414  }
415  else
416  {
417  log << "Only energies will used for training.\n";
418  }
419 
420  if (settings.keywordExists("main_error_metric"))
421  {
422  if (settings["main_error_metric"] == "RMSEpa")
423  {
424  errorMetricEnergies = 0;
425  errorMetricForces = 0;
426  }
427  else if (settings["main_error_metric"] == "RMSE")
428  {
429  errorMetricEnergies = 1;
430  errorMetricForces = 0;
431  }
432  else if (settings["main_error_metric"] == "MAEpa")
433  {
434  errorMetricEnergies = 2;
435  errorMetricForces = 1;
436  }
437  else if (settings["main_error_metric"] == "MAE")
438  {
439  errorMetricEnergies = 3;
440  errorMetricForces = 1;
441  }
442  else
443  {
444  throw runtime_error("ERROR: Unknown error metric.\n");
445  }
446  }
447  else
448  {
449  errorMetricEnergies = 0;
450  errorMetricForces = 0;
451  }
452 
453  updaterType = (UpdaterType)atoi(settings["updater_type"].c_str());
454  if (updaterType == UT_GD)
455  {
456  log << strpr("Weight update via gradient descent selected: "
457  "updaterType::UT_GD (%d)\n",
458  updaterType);
459  }
460  else if (updaterType == UT_KF)
461  {
462  log << strpr("Weight update via Kalman filter selected: "
463  "updaterType::UT_KF (%d)\n",
464  updaterType);
465  }
466  else if (updaterType == UT_LM)
467  {
468  throw runtime_error("ERROR: LM algorithm not yet implemented.\n");
469  log << strpr("Weight update via Levenberg-Marquardt algorithm "
470  "selected: updaterType::UT_LM (%d)\n",
471  updaterType);
472  }
473  else
474  {
475  throw runtime_error("ERROR: Unknown updater type.\n");
476  }
477 
478  parallelMode = (ParallelMode)atoi(settings["parallel_mode"].c_str());
479  //if (parallelMode == PM_DATASET)
480  //{
481  // log << strpr("Serial training selected: "
482  // "ParallelMode::PM_DATASET (%d)\n",
483  // parallelMode);
484  //}
485  if (parallelMode == PM_TRAIN_RK0)
486  {
487  log << strpr("Parallel training (rank 0 updates) selected: "
488  "ParallelMode::PM_TRAIN_RK0 (%d)\n",
489  parallelMode);
490  }
491  else if (parallelMode == PM_TRAIN_ALL)
492  {
493  log << strpr("Parallel training (all ranks update) selected: "
494  "ParallelMode::PM_TRAIN_ALL (%d)\n",
495  parallelMode);
496  }
497  else
498  {
499  throw runtime_error("ERROR: Unknown parallelization mode.\n");
500  }
501 
502  jacobianMode = (JacobianMode)atoi(settings["jacobian_mode"].c_str());
503  if (jacobianMode == JM_SUM)
504  {
505  log << strpr("Gradient summation only selected: "
506  "JacobianMode::JM_SUM (%d)\n", jacobianMode);
507  log << "No Jacobi matrix, gradients of all training candidates are "
508  "summed up instead.\n";
509  }
510  else if (jacobianMode == JM_TASK)
511  {
512  log << strpr("Per-task Jacobian selected: "
513  "JacobianMode::JM_TASK (%d)\n",
514  jacobianMode);
515  log << "One Jacobi matrix row per MPI task is stored, within each "
516  "task gradients are summed up.\n";
517  }
518  else if (jacobianMode == JM_FULL)
519  {
520  log << strpr("Full Jacobian selected: "
521  "JacobianMode::JM_FULL (%d)\n",
522  jacobianMode);
523  log << "Each update candidate generates one Jacobi matrix "
524  "row entry.\n";
525  }
526  else
527  {
528  throw runtime_error("ERROR: Unknown Jacobian mode.\n");
529  }
530 
531  if (updaterType == UT_GD && jacobianMode != JM_SUM)
532  {
533  throw runtime_error("ERROR: Gradient descent methods can only be "
534  "combined with Jacobian mode JM_SUM.\n");
535  }
536 
537  updateStrategy = (UpdateStrategy)atoi(settings["update_strategy"].c_str());
538  // This section is pushed into a separate function because it's needed also
539  // for testing purposes.
540  initializeWeightsMemory(updateStrategy);
541  // Now it is possible to fill the weights arrays with weight parameters
542  // from the neural network.
543  getWeights();
544 
545  vector<string> selectionModeArgs = split(settings["selection_mode"]);
546  if (selectionModeArgs.size() % 2 != 1)
547  {
548  throw runtime_error("ERROR: Incorrect selection mode format.\n");
549  }
550  selectionModeSchedule[0] =
551  (SelectionMode)atoi(selectionModeArgs.at(0).c_str());
552  for (size_t i = 1; i < selectionModeArgs.size(); i = i + 2)
553  {
554  selectionModeSchedule[(size_t)atoi(selectionModeArgs.at(i).c_str())] =
555  (SelectionMode)atoi(selectionModeArgs.at(i + 1).c_str());
556  }
557  for (map<size_t, SelectionMode>::const_iterator it =
558  selectionModeSchedule.begin();
559  it != selectionModeSchedule.end(); ++it)
560  {
561  log << strpr("Selection mode starting with epoch %zu:\n", it->first);
562  if (it->second == SM_RANDOM)
563  {
564  log << strpr("Random selection of update candidates: "
565  "SelectionMode::SM_RANDOM (%d)\n", it->second);
566  }
567  else if (it->second == SM_SORT)
568  {
569  log << strpr("Update candidates selected according to error: "
570  "SelectionMode::SM_SORT (%d)\n", it->second);
571  }
572  else if (it->second == SM_THRESHOLD)
573  {
574  log << strpr("Update candidates chosen randomly above RMSE "
575  "threshold: SelectionMode::SM_THRESHOLD (%d)\n",
576  it->second);
577  rmseThresholdEnergy
578  = atof(settings["short_energy_error_threshold"].c_str());
579  rmseThresholdForce
580  = atof(settings["short_force_error_threshold"].c_str());
581  rmseThresholdTrials
582  = atof(settings["rmse_threshold_trials"].c_str());
583  log << strpr("Energy threshold: %.2f * RMSE(Energy)\n",
584  rmseThresholdEnergy);
585  if (useForces)
586  {
587  log << strpr("Force threshold: %.2f * RMSE(Force)\n",
588  rmseThresholdForce);
589  }
590  log << strpr("Maximum number of update candidate trials: %zu\n",
591  rmseThresholdTrials);
592  }
593  else
594  {
595  throw runtime_error("ERROR: Unknown selection mode.\n");
596  }
597  }
598  selectionMode = selectionModeSchedule[0];
599 
600  log << "-----------------------------------------"
601  "--------------------------------------\n";
602  reapeatedEnergyUpdates = settings.keywordExists("repeated_energy_update");
603  if (useForces && reapeatedEnergyUpdates)
604  {
605  throw runtime_error("ERROR: Repeated energy updates are not correctly"
606  " implemented at the moment.\n");
607  //log << "After each force update an energy update for the\n";
608  //log << "corresponding structure will be performed.\n";
609  }
610 
611  freeMemory = !(settings.keywordExists("memorize_symfunc_results"));
612  if (freeMemory)
613  {
614  log << "Symmetry function memory is cleared after each calculation.\n";
615  }
616  else
617  {
618  log << "Symmetry function memory is reused (HIGH MEMORY USAGE!).\n";
619  }
620 
621  numEpochs = (size_t)atoi(settings["epochs"].c_str());
622  log << strpr("Training will be stopped after %zu epochs.\n", numEpochs);
623 
624  // Check how often energy comparison files should be written.
625  if (settings.keywordExists("write_trainpoints"))
626  {
627  writeEnergiesEvery = 1;
628  vector<string> v = split(reduce(settings["write_trainpoints"]));
629  if (v.size() == 1)
630  {
631  writeEnergiesEvery = (size_t)atoi(v.at(0).c_str());
632  }
633  else if (v.size() == 2)
634  {
635  writeEnergiesEvery = (size_t)atoi(v.at(0).c_str());
636  writeEnergiesAlways = (size_t)atoi(v.at(1).c_str());
637  }
638  log << strpr("Energy comparison will be written every %d epochs.\n",
639  writeEnergiesEvery);
640  if (writeEnergiesAlways > 0)
641  {
642  log << strpr("Up to epoch %d energy comparison will be written"
643  " every epoch.\n", writeEnergiesAlways);
644  }
645  }
646 
647  // Check how often force comparison files should be written.
648  if (settings.keywordExists("write_trainforces"))
649  {
650  writeForcesEvery = 1;
651  vector<string> v = split(reduce(settings["write_trainforces"]));
652  if (v.size() == 1)
653  {
654  writeForcesEvery = (size_t)atoi(v.at(0).c_str());
655  }
656  else if (v.size() == 2)
657  {
658  writeForcesEvery = (size_t)atoi(v.at(0).c_str());
659  writeForcesAlways = (size_t)atoi(v.at(1).c_str());
660  }
661  if (useForces)
662  {
663  log << strpr("Force comparison will be written every %d epochs.\n",
664  writeForcesEvery);
665  }
666  if (writeForcesAlways > 0 && useForces)
667  {
