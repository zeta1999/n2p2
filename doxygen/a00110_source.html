108 double Stopwatch::updateTime()
109 {
110  if (state == RUNNING)
111  {
112 #ifdef NOTIME
113  return 1.0;
114 #elif __linux__
115  time_t secLast = time.tv_sec;
116  long nsecLast = time.tv_nsec;
117 
118  clock_gettime(CLOCK_MONOTONIC, &time);
119  return (double)(time.tv_sec - secLast)
120  + (double)(time.tv_nsec - nsecLast) * NSEC;
121 #elif __MACH__
122  uint64_t lastTime = time;
123  uint64_t time = mach_absolute_time();
124  static mach_timebase_info_data_t info = {0, 0};
125 
126  if (info.denom == 0) mach_timebase_info(&info);
127  return ((time - lastTime) * (info.numer / info.denom)) * NSEC;
128 #endif
129  }
130  else
131  {
132  fprintf(stderr,
133  "WARNING: Unable to update time, clock not running.\n");
134  return 0.0;
135  }
136 }
