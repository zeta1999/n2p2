85 bool SymmetryFunctionGroupAngularWide::
86 addMember(SymmetryFunction const* const symmetryFunction)
87 {
88  if (symmetryFunction->getType() != type) return false;
89 
90  SymmetryFunctionAngularWide const* sf =
91  dynamic_cast<SymmetryFunctionAngularWide const*>(
92  symmetryFunction);
93 
94  if (members.empty())
95  {
96  cutoffType = sf->getCutoffType();
97  cutoffAlpha = sf->getCutoffAlpha();
98  ec = sf->getEc();
99  rc = sf->getRc();
100  e1 = sf->getE1();
101  e2 = sf->getE2();
102  convLength = sf->getConvLength();
103 
104  fc.setCutoffType(cutoffType);
105  fc.setCutoffRadius(rc);
106  fc.setCutoffParameter(cutoffAlpha);
107  }
108 
109  if (sf->getCutoffType() != cutoffType ) return false;
110  if (sf->getCutoffAlpha() != cutoffAlpha) return false;
111  if (sf->getEc() != ec ) return false;
112  if (sf->getRc() != rc ) return false;
113  if (sf->getE1() != e1 ) return false;
114  if (sf->getE2() != e2 ) return false;
115  if (sf->getConvLength() != convLength )
116  {
117  throw runtime_error("ERROR: Unable to add symmetry function members "
118  "with different conversion factors.\n");
119  }
120 
121  members.push_back(sf);
122 
123  return true;
124 }

126 void SymmetryFunctionGroupAngularWide::sortMembers()
127 {
128  sort(members.begin(),
129  members.end(),
130  comparePointerTargets<SymmetryFunctionAngularWide const>);
131 
132  // Members are now sorted with eta changing the slowest.
133  for (size_t i = 0; i < members.size(); i++)
134  {
135  factorNorm.push_back(pow(2.0, 1.0 - members[i]->getZeta()));
136  factorDeriv.push_back(2.0 * members[i]->getEta() /
137  members[i]->getZeta() / members[i]->getLambda());
138  if (i == 0)
139  {
140  calculateExp.push_back(true);
141  }
142  else
143  {
144  if ( members[i - 1]->getEta() != members[i]->getEta() ||
145  members[i - 1]->getRs() != members[i]->getRs() )
146  {
147  calculateExp.push_back(true);
148  }
149  else
150  {
151  calculateExp.push_back(false);
152  }
153  }
154  useIntegerPow.push_back(members[i]->getUseIntegerPow());
155  memberIndex.push_back(members[i]->getIndex());
156  zetaInt.push_back(members[i]->getZetaInt());
157  eta.push_back(members[i]->getEta());
158  rs.push_back(members[i]->getRs());
159  lambda.push_back(members[i]->getLambda());
160  zeta.push_back(members[i]->getZeta());
161  zetaLambda.push_back(members[i]->getZeta() * members[i]->getLambda());
162  memberIndexPerElement.push_back(members[i]->getIndexPerElement());
163  }
164 
165  return;
166 }

180 // Depending on chosen symmetry functions this function may be very
181 // time-critical when predicting new structures (e.g. in MD simulations). Thus,
182 // lots of optimizations were used sacrificing some readablity. Vec3D
183 // operations have been rewritten in simple C array style and the use of
184 // temporary objects has been minimized. Some of the originally coded
185 // expressions are kept in comments marked with "SIMPLE EXPRESSIONS:".
186 void SymmetryFunctionGroupAngularWide::calculate(Atom& atom,
187  bool const derivatives) const
188 {
189  double* result = new double[members.size()];
190  for (size_t l = 0; l < members.size(); ++l)
191  {
192  result[l] = 0.0;
193  }
194 
195  size_t numNeighbors = atom.numNeighbors;
196  // Prevent problematic condition in loop test below (j < numNeighbors - 1).
197  if (numNeighbors == 0) numNeighbors = 1;
198 
199  for (size_t j = 0; j < numNeighbors - 1; j++)
200  {
201  Atom::Neighbor& nj = atom.neighbors[j];
202  size_t const nej = nj.element;
203  double const rij = nj.d;
204  if ((e1 == nej || e2 == nej) && rij < rc)
205  {
206  double const r2ij = rij * rij;
207 
208  // Calculate cutoff function and derivative.
209 #ifdef NOCFCACHE
210  double pfcij;
211  double pdfcij;
212  fc.fdf(rij, pfcij, pdfcij);
213 #else
214  // If cutoff radius matches with the one in the neighbor storage
215  // we can use the previously calculated value.
216  double& pfcij = nj.fc;
217  double& pdfcij = nj.dfc;
218  if (nj.cutoffType != cutoffType ||
219  nj.rc != rc ||
220  nj.cutoffAlpha != cutoffAlpha)
221  {
222  fc.fdf(rij, pfcij, pdfcij);
223  nj.rc = rc;
224  nj.cutoffType = cutoffType;
225  nj.cutoffAlpha = cutoffAlpha;
226  }
227 #endif
228  // SIMPLE EXPRESSIONS:
229  //Vec3D drij(atom.neighbors[j].dr);
230  double const* const dr1 = nj.dr.r;
231 
232  for (size_t k = j + 1; k < numNeighbors; k++)
233  {
234  Atom::Neighbor& nk = atom.neighbors[k];
235  size_t const nek = nk.element;
236  if ((e1 == nej && e2 == nek) ||
237  (e2 == nej && e1 == nek))
238  {
239  double const rik = nk.d;
240  if (rik < rc)
241  {
242  // SIMPLE EXPRESSIONS:
243  //Vec3D drik(atom.neighbors[k].dr);
244  //Vec3D drjk = drik - drij;
245  double const* const dr2 = nk.dr.r;
246  double const dr30 = dr2[0] - dr1[0];
247  double const dr31 = dr2[1] - dr1[1];
248  double const dr32 = dr2[2] - dr1[2];
249 
250  // Energy calculation.
251 #ifdef NOCFCACHE
252  double pfcik;
253  double pdfcik;
254  fc.fdf(rik, pfcik, pdfcik);
255 #else
256  double& pfcik = nk.fc;
257  double& pdfcik = nk.dfc;
258  if (nk.cutoffType != cutoffType ||
259  nk.rc != rc ||
260  nk.cutoffAlpha != cutoffAlpha)
261  {
262  fc.fdf(rik, pfcik, pdfcik);
263  nk.rc = rc;
264  nk.cutoffType = cutoffType;
265  nk.cutoffAlpha = cutoffAlpha;
266  }
267 #endif
268  double const rinvijik = 1.0 / rij / rik;
269  // SIMPLE EXPRESSIONS:
270  //double const costijk = drij * drik * rinvijik;
271  double const costijk = (dr1[0] * dr2[0] +
272  dr1[1] * dr2[1] +
273  dr1[2] * dr2[2]) * rinvijik;
274  double const pfc = pfcij * pfcik;
275  double const r2ik = rik * rik;
276  double const r2sum = r2ij + r2ik;
277  double const pr1 = pfcik * pdfcij / rij;
278  double const pr2 = pfcij * pdfcik / rik;
279  double vexp = 0.0;
280  double rijs = 0.0;
281  double riks = 0.0;
282 
283  for (size_t l = 0; l < members.size(); ++l)
284  {
285  if (calculateExp[l])
286  {
287  if (rs[l] > 0.0)
288  {
289  rijs = rij - rs[l];
290  riks = rik - rs[l];
291  vexp = exp(-eta[l] * (rijs * rijs
292  + riks * riks));
293  }
294  else
295  {
296  vexp = exp(-eta[l] * r2sum);
297  }
298  }
299  double const plambda = 1.0 + lambda[l] * costijk;
300  double fg = vexp;
301  if (plambda <= 0.0) fg = 0.0;
302  else
303  {
304  if (useIntegerPow[l])
305  {
306  fg *= pow_int(plambda, zetaInt[l] - 1);
307  }
308  else
309  {
310  fg *= pow(plambda, zeta[l] - 1.0);
311  }
312  }
313  result[l] += fg * plambda * pfc;
314 
315  // Force calculation.
316  if (!derivatives) continue;
317  fg *= factorNorm[l];
318  double const pfczl = pfc * zetaLambda[l];
319  double const p2etapl = plambda * factorDeriv[l];
320  double p1;
321  double p2;
322  if (rs[l] > 0)
323  {
324  p1 = fg * (pfczl * (rinvijik
325  - costijk / r2ij - p2etapl
326  * rijs / rij) + pr1 * plambda);
327  p2 = fg * (pfczl * (rinvijik
328  - costijk / r2ik - p2etapl
329  * riks / rik) + pr2 * plambda);
330  }
331  else
332  {
333  p1 = fg * (pfczl * (rinvijik - costijk / r2ij
334  - p2etapl) + pr1 * plambda);
335  p2 = fg * (pfczl * (rinvijik - costijk / r2ik
336  - p2etapl) + pr2 * plambda);
337 
338  }
339  double const p3 = fg * pfczl * rinvijik;
340 
341  // SIMPLE EXPRESSIONS:
342  // Save force contributions in Atom storage.
343  //atom.dGdr[memberIndex[l]] += p1 * drij
344  // + p2 * drik;
345  //atom.neighbors[j].
346  // dGdr[memberIndex[l]] -= p1 * drij
347  // + p3 * drjk;
348  //atom.neighbors[k].
349  // dGdr[memberIndex[l]] -= p2 * drik
350  // - p3 * drjk;
351 
352  double const p1drijx = p1 * dr1[0];
353  double const p1drijy = p1 * dr1[1];
354  double const p1drijz = p1 * dr1[2];
355 
356  double const p2drikx = p2 * dr2[0];
357  double const p2driky = p2 * dr2[1];
358  double const p2drikz = p2 * dr2[2];
359 
360  double const p3drjkx = p3 * dr30;
361  double const p3drjky = p3 * dr31;
362  double const p3drjkz = p3 * dr32;
363 
364 #ifdef IMPROVED_SFD_MEMORY
365  size_t li = memberIndex[l];
366 #else
367  size_t const li = memberIndex[l];
368 #endif
369  double* dGdr = atom.dGdr[li].r;
370  dGdr[0] += p1drijx + p2drikx;
371  dGdr[1] += p1drijy + p2driky;
372  dGdr[2] += p1drijz + p2drikz;
373 
374 #ifdef IMPROVED_SFD_MEMORY
375  li = memberIndexPerElement[l][nej];
376 #endif
377  dGdr = nj.dGdr[li].r;
378  dGdr[0] -= p1drijx + p3drjkx;
379  dGdr[1] -= p1drijy + p3drjky;
380  dGdr[2] -= p1drijz + p3drjkz;
381 
382 #ifdef IMPROVED_SFD_MEMORY
383  li = memberIndexPerElement[l][nek];
384 #endif
385  dGdr = nk.dGdr[li].r;
386  dGdr[0] -= p2drikx - p3drjkx;
387  dGdr[1] -= p2driky - p3drjky;
388  dGdr[2] -= p2drikz - p3drjkz;
389  } // l
390  } // rik <= rc
391  } // elem
392  } // k
393  } // rij <= rc
394  } // j
395 
396  for (size_t l = 0; l < members.size(); ++l)
397  {
398  result[l] *= factorNorm[l] / scalingFactors[l];
399  atom.G[memberIndex[l]] = members[l]->scale(result[l]);
400  }
401 
402  delete[] result;
403 
404  return;
405 }
