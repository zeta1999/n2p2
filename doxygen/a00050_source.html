40 void Atom::collectDGdxia(size_t indexAtom, size_t indexComponent)
41 {
42  for (size_t i = 0; i < dGdxia.size(); i++)
43  {
44  dGdxia[i] = 0.0;
45  }
46  for (size_t i = 0; i < numNeighbors; i++)
47  {
48  if (neighbors[i].index == indexAtom)
49  {
50  for (size_t j = 0; j < numSymmetryFunctions; ++j)
51  {
52  dGdxia[j] += neighbors[i].dGdr[j][indexComponent];
53  }
54  }
55  }
56  if (index == indexAtom)
57  {
58  for (size_t i = 0; i < numSymmetryFunctions; ++i)
59  {
60  dGdxia[i] += dGdr[i][indexComponent];
61  }
62  }
63 
64  return;
65 }

68 void Atom::toNormalizedUnits(double convEnergy, double convLength)
69 {
70  energy *= convEnergy;
71  r *= convLength;
72  f *= convEnergy / convLength;
73  fRef *= convEnergy / convLength;
74  if (hasSymmetryFunctionDerivatives)
75  {
76  for (size_t i = 0; i < numSymmetryFunctions; ++i)
77  {
78  dEdG.at(i) *= convEnergy;
79  dGdr.at(i) /= convLength;
80 #ifndef IMPROVED_SFD_MEMORY
81  dGdxia.at(i) /= convLength;
82 #endif
83  }
84  }
85 
86  if (hasNeighborList)
87  {
88  for (vector<Neighbor>::iterator it = neighbors.begin();
89  it != neighbors.end(); ++it)
90  {
91  it->d *= convLength;
92  it->dfc /= convLength;
93  it->rc *= convLength;
94  it->dr *= convLength;
95  if (hasSymmetryFunctionDerivatives)
96  {
97  for (size_t i = 0; i < dGdr.size(); ++i)
98  {
99  dGdr.at(i) /= convLength;
100  }
101  }
102  }
103  }
104 
105  return;

148 void Atom::allocate(bool all)
149 {
150  if (numSymmetryFunctions == 0)
151  {
152  throw range_error("ERROR: Number of symmetry functions set to"
153  "zero, cannot allocate.\n");
154  }
155 
156  // Clear all symmetry function related vectors (also for derivatives).
157  G.clear();
158  dEdG.clear();
159 #ifndef IMPROVED_SFD_MEMORY
160  dGdxia.clear();
161 #endif
162  dGdr.clear();
163  for (vector<Neighbor>::iterator it = neighbors.begin();
164  it != neighbors.end(); ++it)
165  {
166  it->dGdr.clear();
167  }
168 
169  // Reset status of symmetry functions and derivatives.
170  hasSymmetryFunctions = false;
171  hasSymmetryFunctionDerivatives = false;
172 
173  // Resize vectors (derivatives only if requested).
174  G.resize(numSymmetryFunctions, 0.0);
175  if (all)
176  {
177 #ifdef IMPROVED_SFD_MEMORY
178  if (numSymmetryFunctionDerivatives.size() == 0)
179  {
180  throw range_error("ERROR: Number of symmetry function derivatives"
181  " unset, cannot allocate.\n");
182  }
183 #endif
184  dEdG.resize(numSymmetryFunctions, 0.0);
185 #ifndef IMPROVED_SFD_MEMORY
186  dGdxia.resize(numSymmetryFunctions, 0.0);
187 #endif
188  dGdr.resize(numSymmetryFunctions);
189  for (vector<Neighbor>::iterator it = neighbors.begin();
190  it != neighbors.end(); ++it)
191  {
192 #ifdef IMPROVED_SFD_MEMORY
193  it->dGdr.resize(numSymmetryFunctionDerivatives.at(it->element));
194 #else
195  it->dGdr.resize(numSymmetryFunctions);
196 #endif
197  }
198  }
199 
200  return;
201 }
