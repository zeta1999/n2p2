73 void Element::addSymmetryFunction(string const& parameters,
74  size_t const& lineNumber)
75 {
76  vector<string> args = split(reduce(parameters));
77  size_t type = (size_t)atoi(args.at(1).c_str());
78 
79  if (type == 2)
80  {
81  symmetryFunctions.push_back(
82  new SymmetryFunctionRadial(elementMap));
83  }
84  else if (type == 3)
85  {
86  symmetryFunctions.push_back(
87  new SymmetryFunctionAngularNarrow(elementMap));
88  }
89  else if (type == 9)
90  {
91  symmetryFunctions.push_back(
92  new SymmetryFunctionAngularWide(elementMap));
93  }
94  else if (type == 12)
95  {
96  symmetryFunctions.push_back(
97  new SymmetryFunctionWeightedRadial(elementMap));
98  }
99  else if (type == 13)
100  {
101  symmetryFunctions.push_back(
102  new SymmetryFunctionWeightedAngular(elementMap));
103  }
104  else
105  {
106  throw runtime_error("ERROR: Unknown symmetry function type.\n");
107  }
108 
109  symmetryFunctions.back()->setParameters(parameters);
110  symmetryFunctions.back()->setLineNumber(lineNumber);
111 
112  return;

166 void Element::setupSymmetryFunctionGroups()
167 {
168  for (vector<SymmetryFunction*>::const_iterator
169  sf = symmetryFunctions.begin(); sf != symmetryFunctions.end(); ++sf)
170  {
171  bool createNewGroup = true;
172  for (vector<SymmetryFunctionGroup*>::const_iterator
173  sfg = symmetryFunctionGroups.begin();
174  sfg != symmetryFunctionGroups.end(); ++sfg)
175  {
176  if ((*sfg)->addMember((*sf)))
177  {
178  createNewGroup = false;
179  break;
180  }
181  }
182  if (createNewGroup)
183  {
184  if ((*sf)->getType() == 2)
185  {
186  symmetryFunctionGroups.push_back((SymmetryFunctionGroup*)
187  new SymmetryFunctionGroupRadial(elementMap));
188  }
189  else if ((*sf)->getType() == 3)
190  {
191  symmetryFunctionGroups.push_back((SymmetryFunctionGroup*)
192  new SymmetryFunctionGroupAngularNarrow(elementMap));
193  }
194  else if ((*sf)->getType() == 9)
195  {
196  symmetryFunctionGroups.push_back((SymmetryFunctionGroup*)
197  new SymmetryFunctionGroupAngularWide(elementMap));
198  }
199  else if ((*sf)->getType() == 12)
200  {
201  symmetryFunctionGroups.push_back((SymmetryFunctionGroup*)
202  new SymmetryFunctionGroupWeightedRadial(elementMap));
203  }
204  else if ((*sf)->getType() == 13)
205  {
206  symmetryFunctionGroups.push_back((SymmetryFunctionGroup*)
207  new SymmetryFunctionGroupWeightedAngular(elementMap));
208  }
209  else
210  {
211  throw runtime_error("ERROR: Unknown symmetry function group"
212  " type.\n");
213  }
214  symmetryFunctionGroups.back()->addMember(*sf);
215  }
216  }
217 
218  sort(symmetryFunctionGroups.begin(),
219  symmetryFunctionGroups.end(),
220  comparePointerTargets<SymmetryFunctionGroup>);
221 
222  for (size_t i = 0; i < symmetryFunctionGroups.size(); ++i)
223  {
224  symmetryFunctionGroups.at(i)->sortMembers();
225  symmetryFunctionGroups.at(i)->setIndex(i);
226  }
227 
228  return;
229 }

241  s->setIndexPerElement(i, symmetryFunctionTable.at(i).size());
242  symmetryFunctionTable.at(i).push_back(s->getIndex());
243  }
244  }
245  }
246  for (size_t i = 0; i < elementMap.size(); ++i)
247  {
248  symmetryFunctionNumTable.push_back(symmetryFunctionTable.at(i).size());
249  }
250

386 size_t Element::updateSymmetryFunctionStatistics(Atom const& atom)
387 {
388  size_t countExtrapolationWarnings = 0;
389  double epsilon = 10.0 * numeric_limits<double>::epsilon();
390 
391  if (atom.element != index)
392  {
393  throw runtime_error("ERROR: Atom has a different element index.\n");
394  }
395 
396  if (atom.numSymmetryFunctions != symmetryFunctions.size())
397  {
398  throw runtime_error("ERROR: Number of symmetry functions"
399  " does not match.\n");
400  }
401 
402  for (size_t i = 0; i < atom.G.size(); ++i)
403  {
404  double const Gmin = symmetryFunctions.at(i)->getGmin();
405  double const Gmax = symmetryFunctions.at(i)->getGmax();
406  double const value = symmetryFunctions.at(i)->unscale(atom.G.at(i));
407  size_t const index = symmetryFunctions.at(i)->getIndex();
408  if (statistics.collectStatistics)
409  {
410  statistics.addValue(index, atom.G.at(i));
411  }
412 
413  // Avoid "fake" EWs at the boundaries.
414  if (value + epsilon < Gmin || value - epsilon > Gmax)
415  {
416  countExtrapolationWarnings++;
417  if (statistics.collectExtrapolationWarnings)
418  {
419  statistics.addExtrapolationWarning(index,
420  value,
421  Gmin,
422  Gmax,
423  atom.indexStructure,
424  atom.tag);
425  }
426  if (statistics.writeExtrapolationWarnings)
427  {
428  cerr << strpr("### NNP EXTRAPOLATION WARNING ### "
429  "STRUCTURE: %6zu ATOM: %6zu SYMFUNC: %4zu "
430  "VALUE: %10.3E MIN: %10.3E MAX: %10.3E\n",
431  atom.indexStructure,
432  atom.tag,
433  index,
434  value,
435  Gmin,
436  Gmax);
437  }
438  if (statistics.stopOnExtrapolationWarnings)
439  {
440  throw out_of_range(
441  strpr("### NNP EXTRAPOLATION WARNING ### "
442  "STRUCTURE: %6zu ATOM: %6zu SYMFUNC: %4zu "
443  "VALUE: %10.3E MIN: %10.3E MAX: %10.3E\n"
444  "ERROR: Symmetry function value out of range.\n",
445  atom.indexStructure,
446  atom.tag,
447  index,
448  value,
449  Gmin,
450  Gmax));
451  }
452  }
453  }
454 
455  return countExtrapolationWarnings;
456 }
