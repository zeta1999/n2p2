85 bool SymmetryFunctionGroupAngularNarrow::
86 addMember(SymmetryFunction const* const symmetryFunction)
87 {
88  if (symmetryFunction->getType() != type) return false;

90  SymmetryFunctionAngularNarrow const* sf =

92  symmetryFunction);
93 
94  if (members.empty())
95  {
96  cutoffType = sf->getCutoffType();
97  cutoffAlpha = sf->getCutoffAlpha();
98  ec = sf->getEc();
99  rc = sf->getRc();
100  e1 = sf->getE1();
101  e2 = sf->getE2();
102  convLength = sf->getConvLength();
103 
104  fc.setCutoffType(cutoffType);
105  fc.setCutoffRadius(rc);
106  fc.setCutoffParameter(cutoffAlpha);
107  }
108 
109  if (sf->getCutoffType() != cutoffType ) return false;
110  if (sf->getCutoffAlpha() != cutoffAlpha) return false;
111  if (sf->getEc() != ec ) return false;
112  if (sf->getRc() != rc ) return false;
113  if (sf->getE1() != e1 ) return false;
114  if (sf->getE2() != e2 ) return false;
115  if (sf->getConvLength() != convLength )
116  {
117  throw runtime_error("ERROR: Unable to add symmetry function members "
118  "with different conversion factors.\n");
119  }
120 
121  members.push_back(sf);
122 
123  return true;

126 void SymmetryFunctionGroupAngularNarrow::sortMembers()
127 {
128  sort(members.begin(),
129  members.end(),
130  comparePointerTargets<SymmetryFunctionAngularNarrow const>);
131 
132  // Members are now sorted with eta changing the slowest.
133  for (size_t i = 0; i < members.size(); i++)
134  {
135  factorNorm.push_back(pow(2.0, 1.0 - members[i]->getZeta()));
136  factorDeriv.push_back(2.0 * members[i]->getEta() /
137  members[i]->getZeta() / members[i]->getLambda());
138  if (i == 0)
139  {
140  calculateExp.push_back(true);
141  }
142  else
143  {
144  if ( members[i - 1]->getEta() != members[i]->getEta() ||
145  members[i - 1]->getRs() != members[i]->getRs() )
146  {
147  calculateExp.push_back(true);
148  }
149  else
150  {
151  calculateExp.push_back(false);
152  }
153  }
154  useIntegerPow.push_back(members[i]->getUseIntegerPow());
155  memberIndex.push_back(members[i]->getIndex());
156  zetaInt.push_back(members[i]->getZetaInt());
157  eta.push_back(members[i]->getEta());
158  rs.push_back(members[i]->getRs());
159  zeta.push_back(members[i]->getZeta());
160  lambda.push_back(members[i]->getLambda());
161  zetaLambda.push_back(members[i]->getZeta() * members[i]->getLambda());
162  memberIndexPerElement.push_back(members[i]->getIndexPerElement());
163  }
164 
165  return;
166 }

180 // Depending on chosen symmetry functions this function may be very
181 // time-critical when predicting new structures (e.g. in MD simulations). Thus,
182 // lots of optimizations were used sacrificing some readablity. Vec3D
183 // operations have been rewritten in simple C array style and the use of
184 // temporary objects has been minimized. Some of the originally coded
185 // expressions are kept in comments marked with "SIMPLE EXPRESSIONS:".
186 void SymmetryFunctionGroupAngularNarrow::calculate(
187  Atom& atom,
188  bool const derivatives) const
189 {
190  double* result = new double[members.size()];
191  for (size_t l = 0; l < members.size(); ++l)
192  {
193  result[l] = 0.0;
194  }
195 
196  double const rc2 = rc * rc;
197  size_t numNeighbors = atom.numNeighbors;
198  // Prevent problematic condition in loop test below (j < numNeighbors - 1).
199  if (numNeighbors == 0) numNeighbors = 1;
200 
201  for (size_t j = 0; j < numNeighbors - 1; j++)
202  {
203  Atom::Neighbor& nj = atom.neighbors[j];
204  size_t const nej = nj.element;
205  double const rij = nj.d;
206  if ((e1 == nej || e2 == nej) && rij < rc)
207  {
208  double const r2ij = rij * rij;
209 
210  // Calculate cutoff function and derivative.
211 #ifdef NOCFCACHE
212  double pfcij;
213  double pdfcij;
214  fc.fdf(rij, pfcij, pdfcij);
215 #else
216  // If cutoff radius matches with the one in the neighbor storage
217  // we can use the previously calculated value.
218  double& pfcij = nj.fc;
219  double& pdfcij = nj.dfc;
220  if (nj.cutoffType != cutoffType ||
221  nj.rc != rc ||
222  nj.cutoffAlpha != cutoffAlpha)
223  {
224  fc.fdf(rij, pfcij, pdfcij);
225  nj.rc = rc;
226  nj.cutoffType = cutoffType;
227  nj.cutoffAlpha = cutoffAlpha;
228  }
229 #endif
230  // SIMPLE EXPRESSIONS:
231  //Vec3D const drij(atom.neighbors[j].dr);
232  //double const* const dr1 = drij.r;
233  double const* const dr1 = nj.dr.r;
234 
235  for (size_t k = j + 1; k < numNeighbors; k++)
236  {
237  Atom::Neighbor& nk = atom.neighbors[k];
238  size_t const nek = nk.element;
239  if ((e1 == nej && e2 == nek) ||
240  (e2 == nej && e1 == nek))
241  {
242  double const rik = nk.d;
243  if (rik < rc)
244  {
245  // SIMPLE EXPRESSIONS:
246  //Vec3D const drjk(atom.neighbors[k].dr
247  // - atom.neighbors[j].dr);
248  //double rjk = drjk.norm2();
249  double const* const dr2 = nk.dr.r;
250  double dr3[3];
251  dr3[0] = dr2[0] - dr1[0];
252  dr3[1] = dr2[1] - dr1[1];
253  dr3[2] = dr2[2] - dr1[2];
254  double rjk = dr3[0] * dr3[0]
255  + dr3[1] * dr3[1]
256  + dr3[2] * dr3[2];
257  if (rjk < rc2)
258  {
259  // Energy calculation.
260 #ifdef NOCFCACHE
261  double pfcik;
262  double pdfcik;
263  fc.fdf(rik, pfcik, pdfcik);
264 #else
265  double& pfcik = nk.fc;
266  double& pdfcik = nk.dfc;
267  if (nk.cutoffType != cutoffType ||
268  nk.rc != rc ||
269  nk.cutoffAlpha != cutoffAlpha)
270  {
271  fc.fdf(rik, pfcik, pdfcik);
272  nk.rc = rc;
273  nk.cutoffType = cutoffType;
274  nk.cutoffAlpha = cutoffAlpha;
275  }
276 #endif
277  rjk = sqrt(rjk);
278 
279  double pfcjk;
280  double pdfcjk;
281  fc.fdf(rjk, pfcjk, pdfcjk);
282 
283  // SIMPLE EXPRESSIONS:
284  //Vec3D const drik(atom.neighbors[k].dr);
285  //double const* const dr2 = drik.r;
286  //double const* const dr3 = drjk.r;
287  double const rinvijik = 1.0 / rij / rik;
288  // SIMPLE EXPRESSIONS:
289  //double const costijk = (drij * drik) * rinvijik;
290  double const costijk = (dr1[0] * dr2[0] +
291  dr1[1] * dr2[1] +
292  dr1[2] * dr2[2]) * rinvijik;
293  double const pfc = pfcij * pfcik * pfcjk;
294  double const r2ik = rik * rik;
295  double const r2sum = r2ij + r2ik + rjk * rjk;
296  double const pr1 = pfcik * pfcjk * pdfcij / rij;
297  double const pr2 = pfcij * pfcjk * pdfcik / rik;
298  double const pr3 = pfcij * pfcik * pdfcjk / rjk;
299  double vexp = 0.0;
300  double rijs = 0.0;
301  double riks = 0.0;
302  double rjks = 0.0;
303  for (size_t l = 0; l < members.size(); ++l)
304  {
305  if (calculateExp[l])
306  {
307  if (rs[l] > 0.0)
308  {
309  rijs = rij - rs[l];
310  riks = rik - rs[l];
311  rjks = rjk - rs[l];
312  vexp = exp(-eta[l] * (rijs * rijs
313  + riks * riks
314  + rjks * rjks));
315  }
316  else
317  {
318  vexp = exp(-eta[l] * r2sum);
319  }
320  }
321  double const plambda = 1.0
322  + lambda[l] * costijk;
323  double fg = vexp;
324  if (plambda <= 0.0) fg = 0.0;
325  else
326  {
327  if (useIntegerPow[l])
328  {
329  fg *= pow_int(plambda, zetaInt[l] - 1);
330  }
331  else
332  {
333  fg *= pow(plambda, zeta[l] - 1.0);
334  }
335  }
336  result[l] += fg * plambda * pfc;
337 
338  // Force calculation.
339  if (!derivatives) continue;
340  fg *= factorNorm[l];
341  double const pfczl = pfc * zetaLambda[l];
342  double const p2etapl = plambda * factorDeriv[l];
343  double p1;
344  double p2;
345  double p3;
346  if (rs[l] > 0.0)
347  {
348  p1 = fg * (pfczl * (rinvijik
349  - costijk / r2ij - p2etapl
350  * rijs / rij) + pr1 * plambda);
351  p2 = fg * (pfczl * (rinvijik
352  - costijk / r2ik - p2etapl
353  * riks / rik) + pr2 * plambda);
354  p3 = fg * (pfczl * (rinvijik
355  + p2etapl * rjks / rjk)
356  - pr3 * plambda);
357  }
358  else
359  {
360  p1 = fg * (pfczl * (rinvijik - costijk
361  / r2ij - p2etapl) + pr1 * plambda);
362  p2 = fg * (pfczl * (rinvijik - costijk
363  / r2ik - p2etapl) + pr2 * plambda);
364  p3 = fg * (pfczl * (rinvijik + p2etapl)
365  - pr3 * plambda);
366  }
367 
368  // Save force contributions in Atom storage.
369  //
370  // SIMPLE EXPRESSIONS:
371  //size_t const li = members[l]->getIndex();
372  //atom.dGdr[li] += p1 * drij + p2 * drik;
373  //atom.neighbors[j].dGdr[li] -= p1 * drij
374  // + p3 * drjk;
375  //atom.neighbors[k].dGdr[li] -= p2 * drik
376  // - p3 * drjk;
377 
378  double const p1drijx = p1 * dr1[0];
379  double const p1drijy = p1 * dr1[1];
380  double const p1drijz = p1 * dr1[2];
381 
382  double const p2drikx = p2 * dr2[0];
383  double const p2driky = p2 * dr2[1];
384  double const p2drikz = p2 * dr2[2];
385 
386  double const p3drjkx = p3 * dr3[0];
387  double const p3drjky = p3 * dr3[1];
388  double const p3drjkz = p3 * dr3[2];
389 
390 #ifdef IMPROVED_SFD_MEMORY
391  size_t li = memberIndex[l];
392 #else
393  size_t const li = memberIndex[l];
394 #endif
395  double* dGdr = atom.dGdr[li].r;
396  dGdr[0] += p1drijx + p2drikx;
397  dGdr[1] += p1drijy + p2driky;
398  dGdr[2] += p1drijz + p2drikz;
399 
400 #ifdef IMPROVED_SFD_MEMORY
401  li = memberIndexPerElement[l][nej];
402 #endif
403  dGdr = nj.dGdr[li].r;
404  dGdr[0] -= p1drijx + p3drjkx;
405  dGdr[1] -= p1drijy + p3drjky;
406  dGdr[2] -= p1drijz + p3drjkz;
407 
408 #ifdef IMPROVED_SFD_MEMORY
409  li = memberIndexPerElement[l][nek];
410 #endif
411  dGdr = nk.dGdr[li].r;
412  dGdr[0] -= p2drikx - p3drjkx;
413  dGdr[1] -= p2driky - p3drjky;
414  dGdr[2] -= p2drikz - p3drjkz;
415  } // l
416  } // rjk <= rc
417  } // rik <= rc
418  } // elem
419  } // k
420  } // rij <= rc
421  } // j
422 
423  for (size_t l = 0; l < members.size(); ++l)
424  {
425  result[l] *= factorNorm[l] / scalingFactors[l];
426  atom.G[memberIndex[l]] = members[l]->scale(result[l]);
427  }
428 
429  delete[] result;
430 
431  return;
432 }
