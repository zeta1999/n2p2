99 void SymmetryFunctionAngularNarrow::
100  setParameters(string const& parameterString)
101 {
102  vector<string> splitLine = split(reduce(parameterString));
103 
104  if (type != (size_t)atoi(splitLine.at(1).c_str()))
105  {
106  throw runtime_error("ERROR: Incorrect symmetry function type.\n");
107  }
108 
109  ec = elementMap[splitLine.at(0)];
110  e1 = elementMap[splitLine.at(2)];
111  e2 = elementMap[splitLine.at(3)];
112  eta = atof(splitLine.at(4).c_str());
113  lambda = atof(splitLine.at(5).c_str());
114  zeta = atof(splitLine.at(6).c_str());
115  rc = atof(splitLine.at(7).c_str());
116  // Shift parameter is optional.
117  if (splitLine.size() > 8)
118  {
119  rs = atof(splitLine.at(8).c_str());
120  }
121 
122  fc.setCutoffRadius(rc);
123  fc.setCutoffParameter(cutoffAlpha);
124 
125  if (e1 > e2)
126  {
127  size_t tmp = e1;
128  e1 = e2;
129  e2 = tmp;
130  }
131 
132  zetaInt = round(zeta);
133  if (fabs(zeta - zetaInt) <= numeric_limits<double>::min())
134  {
135  useIntegerPow = true;
136  }
137  else
138  {
139  useIntegerPow = false;
140  }
141 
142  return;
143 }

175 void SymmetryFunctionAngularNarrow::calculate(Atom& atom,
176  bool const derivatives) const
177 {
178  double const pnorm = pow(2.0, 1.0 - zeta);
179  double const pzl = zeta * lambda;
180  double const rc2 = rc * rc;
181  double result = 0.0;
182 
183  size_t numNeighbors = atom.numNeighbors;
184  // Prevent problematic condition in loop test below (j < numNeighbors - 1).
185  if (numNeighbors == 0) numNeighbors = 1;
186 
187  for (size_t j = 0; j < numNeighbors - 1; j++)
188  {
189  Atom::Neighbor& nj = atom.neighbors[j];
190  size_t const nej = nj.element;
191  double const rij = nj.d;
192  if ((e1 == nej || e2 == nej) && rij < rc)
193  {
194  double const r2ij = rij * rij;
195 
196  // Calculate cutoff function and derivative.
197 #ifdef NOCFCACHE
198  double pfcij;
199  double pdfcij;
200  fc.fdf(rij, pfcij, pdfcij);
201 #else
202  // If cutoff radius matches with the one in the neighbor storage
203  // we can use the previously calculated value.
204  double& pfcij = nj.fc;
205  double& pdfcij = nj.dfc;
206  if (nj.cutoffType != cutoffType ||
207  nj.rc != rc ||
208  nj.cutoffAlpha != cutoffAlpha)
209  {
210  fc.fdf(rij, pfcij, pdfcij);
211  nj.rc = rc;
212  nj.cutoffType = cutoffType;
213  nj.cutoffAlpha = cutoffAlpha;
214  }
215 #endif
216  for (size_t k = j + 1; k < numNeighbors; k++)
217  {
218  Atom::Neighbor& nk = atom.neighbors[k];
219  size_t const nek = nk.element;
220  if ((e1 == nej && e2 == nek) ||
221  (e2 == nej && e1 == nek))
222  {
223  double const rik = nk.d;
224  if (rik < rc)
225  {
226  Vec3D drjk = nk.dr - nj.dr;
227  double rjk = drjk.norm2();;
228  if (rjk < rc2)
229  {
230  // Energy calculation.
231 #ifdef NOCFCACHE
232  double pfcik;
233  double pdfcik;
234  fc.fdf(rik, pfcik, pdfcik);
235 #else
236  double& pfcik = nk.fc;
237  double& pdfcik = nk.dfc;
238  if (nk.cutoffType != cutoffType ||
239  nk.rc != rc ||
240  nk.cutoffAlpha != cutoffAlpha)
241  {
242  fc.fdf(rik, pfcik, pdfcik);
243  nk.rc = rc;
244  nk.cutoffType = cutoffType;
245  nk.cutoffAlpha = cutoffAlpha;
246  }
247 #endif
248  rjk = sqrt(rjk);
249  double pfcjk;
250  double pdfcjk;
251  fc.fdf(rjk, pfcjk, pdfcjk);
252 
253  Vec3D drij = nj.dr;
254  Vec3D drik = nk.dr;
255  double costijk = drij * drik;;
256  double rinvijik = 1.0 / rij / rik;
257  costijk *= rinvijik;
258 
259  double const pfc = pfcij * pfcik * pfcjk;
260  double const r2ik = rik * rik;
261  double const rijs = rij - rs;
262  double const riks = rik - rs;
263  double const rjks = rjk - rs;
264  double const pexp = exp(-eta * (rijs * rijs +
265  riks * riks +
266  rjks * rjks));
267  double const plambda = 1.0 + lambda * costijk;
268  double fg = pexp;
269  if (plambda <= 0.0) fg = 0.0;
270  else
271  {
272  if (useIntegerPow)
273  {
274  fg *= pow_int(plambda, zetaInt - 1);
275  }
276  else
277  {
278  fg *= pow(plambda, zeta - 1.0);
279  }
280  }
281  result += fg * plambda * pfc;
282 
283  // Force calculation.
284  if (!derivatives) continue;
285  fg *= pnorm;
286  rinvijik *= pzl;
287  costijk *= pzl;
288  double const p2etapl = 2.0 * eta * plambda;
289  double const p1 = fg * (pfc * (rinvijik - costijk
290  / r2ij - p2etapl * rijs / rij)
291  + pfcik * pfcjk * pdfcij * plambda
292  / rij);
293  double const p2 = fg * (pfc * (rinvijik - costijk
294  / r2ik - p2etapl * riks / rik)
295  + pfcij * pfcjk * pdfcik * plambda
296  / rik);
297  double const p3 = fg * (pfc * (rinvijik + p2etapl
298  * rjks / rjk) - pfcij * pfcik
299  * pdfcjk * plambda / rjk);
300  drij *= p1 * scalingFactor;
301  drik *= p2 * scalingFactor;
302  drjk *= p3 * scalingFactor;
303 
304  // Save force contributions in Atom storage.
305  atom.dGdr[index] += drij + drik;
306 #ifdef IMPROVED_SFD_MEMORY
307  nj.dGdr[indexPerElement[nej]] -= drij + drjk;
308  nk.dGdr[indexPerElement[nek]] -= drik - drjk;
309 #else
310  nj.dGdr[index] -= drij + drjk;
311  nk.dGdr[index] -= drik - drjk;
312 #endif
313  } // rjk <= rc
314  } // rik <= rc
315  } // elem
316  } // k
317  } // rij <= rc
318  } // j
319  result *= pnorm;
320 
321  atom.G[index] = scale(result);
322 
323  return;
324 }
