123 void SymmetryFunctionWeightedRadial::calculate(Atom& atom,
124  bool const derivatives) const
125 {
126  double result = 0.0;
127 
128  for (size_t j = 0; j < atom.numNeighbors; ++j)
129  {
130  Atom::Neighbor& n = atom.neighbors[j];
131  if (n.d < rc)
132  {
133  // Energy calculation.
134  double const rij = n.d;
135  double const pexp = elementMap.atomicNumber(n.element)
136  * exp(-eta * (rij - rs) * (rij - rs));
137 
138  // Calculate cutoff function and derivative.
139 #ifdef NOCFCACHE
140  double pfc;
141  double pdfc;
142  fc.fdf(rij, pfc, pdfc);
143 #else
144  // If cutoff radius matches with the one in the neighbor storage
145  // we can use the previously calculated value.
146  double& pfc = n.fc;
147  double& pdfc = n.dfc;
148  if (n.cutoffType != cutoffType ||
149  n.rc != rc ||
150  n.cutoffAlpha != cutoffAlpha)
151  {
152  fc.fdf(rij, pfc, pdfc);
153  n.rc = rc;
154  n.cutoffType = cutoffType;
155  n.cutoffAlpha = cutoffAlpha;
156  }
157 #endif
158 
159  result += pexp * pfc;
160  // Force calculation.
161  if (!derivatives) continue;
162  double const p1 = scalingFactor
163  * (pdfc - 2.0 * eta * (rij - rs)
164  * pfc) * pexp / rij;
165  Vec3D dij = p1 * n.dr;
166  // Save force contributions in Atom storage.
167  atom.dGdr[index] += dij;
168 #ifdef IMPROVED_SFD_MEMORY
169  n.dGdr[indexPerElement[n.element]] -= dij;
170 #else
171  n.dGdr[index] -= dij;
172 #endif
173  }
174  }
175 
176  atom.G[index] = scale(result);
177 
178  return;
179 }
