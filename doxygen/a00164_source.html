74 bool SymmetryFunctionGroupWeightedAngular::
75 addMember(SymmetryFunction const* const symmetryFunction)
76 {
77  if (symmetryFunction->getType() != type) return false;
78 
79  SymmetryFunctionWeightedAngular const* sf =
80  dynamic_cast<SymmetryFunctionWeightedAngular const*>(
81  symmetryFunction);
82 
83  if (members.empty())
84  {
85  cutoffType = sf->getCutoffType();
86  cutoffAlpha = sf->getCutoffAlpha();
87  ec = sf->getEc();
88  rc = sf->getRc();
89  convLength = sf->getConvLength();
90 
91  fc.setCutoffType(cutoffType);
92  fc.setCutoffRadius(rc);
93  fc.setCutoffParameter(cutoffAlpha);
94  }
95 
96  if (sf->getCutoffType() != cutoffType ) return false;
97  if (sf->getCutoffAlpha() != cutoffAlpha) return false;
98  if (sf->getEc() != ec ) return false;
99  if (sf->getRc() != rc ) return false;
100  if (sf->getConvLength() != convLength )
101  {
102  throw runtime_error("ERROR: Unable to add symmetry function members "
103  "with different conversion factors.\n");
104  }
105 
106  members.push_back(sf);
107 
108  return true;
109 }

111 void SymmetryFunctionGroupWeightedAngular::sortMembers()
112 {
113  sort(members.begin(),
114  members.end(),
115  comparePointerTargets<SymmetryFunctionWeightedAngular const>);
116 
117  // Members are now sorted with eta changing the slowest.
118  for (size_t i = 0; i < members.size(); i++)
119  {
120  factorNorm.push_back(pow(2.0, 1.0 - members[i]->getZeta()));
121  factorDeriv.push_back(2.0 * members[i]->getEta() /
122  members[i]->getZeta() / members[i]->getLambda());
123  if (i == 0)
124  {
125  calculateExp.push_back(true);
126  }
127  else
128  {
129  if ( members[i - 1]->getEta() != members[i]->getEta() ||
130  members[i - 1]->getRs() != members[i]->getRs() )
131  {
132  calculateExp.push_back(true);
133  }
134  else
135  {
136  calculateExp.push_back(false);
137  }
138  }
139  useIntegerPow.push_back(members[i]->getUseIntegerPow());
140  memberIndex.push_back(members[i]->getIndex());
141  zetaInt.push_back(members[i]->getZetaInt());
142  eta.push_back(members[i]->getEta());
143  rs.push_back(members[i]->getRs());
144  lambda.push_back(members[i]->getLambda());
145  zeta.push_back(members[i]->getZeta());
146  zetaLambda.push_back(members[i]->getZeta() * members[i]->getLambda());
147  memberIndexPerElement.push_back(members[i]->getIndexPerElement());
148  }
149 
150  return;
151 }

165 // Depending on chosen symmetry functions this function may be very
166 // time-critical when predicting new structures (e.g. in MD simulations). Thus,
167 // lots of optimizations were used sacrificing some readablity. Vec3D
168 // operations have been rewritten in simple C array style and the use of
169 // temporary objects has been minimized. Some of the originally coded
170 // expressions are kept in comments marked with "SIMPLE EXPRESSIONS:".
171 void SymmetryFunctionGroupWeightedAngular::calculate(
172  Atom& atom,
173  bool const derivatives) const
174 {
175  double* result = new double[members.size()];
176  for (size_t l = 0; l < members.size(); ++l)
177  {
178  result[l] = 0.0;
179  }
180 
181  double const rc2 = rc * rc;
182  size_t numNeighbors = atom.numNeighbors;
183  // Prevent problematic condition in loop test below (j < numNeighbors - 1).
184  if (numNeighbors == 0) numNeighbors = 1;
185 
186  for (size_t j = 0; j < numNeighbors - 1; j++)
187  {
188  Atom::Neighbor& nj = atom.neighbors[j];
189  size_t const nej = nj.element;
190  double const rij = nj.d;
191  if (rij < rc)
192  {
193  double const r2ij = rij * rij;
194 
195  // Calculate cutoff function and derivative.
196 #ifdef NOCFCACHE
197  double pfcij;
198  double pdfcij;
199  fc.fdf(rij, pfcij, pdfcij);
200 #else
201  // If cutoff radius matches with the one in the neighbor storage
202  // we can use the previously calculated value.
203  double& pfcij = nj.fc;
204  double& pdfcij = nj.dfc;
205  if (nj.cutoffType != cutoffType ||
206  nj.rc != rc ||
207  nj.cutoffAlpha != cutoffAlpha)
208  {
209  fc.fdf(rij, pfcij, pdfcij);
210  nj.rc = rc;
211  nj.cutoffType = cutoffType;
212  nj.cutoffAlpha = cutoffAlpha;
213  }
214 #endif
215  // SIMPLE EXPRESSIONS:
216  //Vec3D const drij(atom.neighbors[j].dr);
217  //double const* const dr1 = drij.r;
218  double const* const dr1 = nj.dr.r;
219 
220  for (size_t k = j + 1; k < numNeighbors; k++)
221  {
222  Atom::Neighbor& nk = atom.neighbors[k];
223  size_t const nek = nk.element;
224  double const rik = nk.d;
225  if (rik < rc)
226  {
227  // SIMPLE EXPRESSIONS:
228  //Vec3D const drjk(atom.neighbors[k].dr
229  // - atom.neighbors[j].dr);
230  //double rjk = drjk.norm2();
231  double const* const dr2 = nk.dr.r;
232  double dr3[3];
233  dr3[0] = dr2[0] - dr1[0];
234  dr3[1] = dr2[1] - dr1[1];
235  dr3[2] = dr2[2] - dr1[2];
236  double rjk = dr3[0] * dr3[0]
237  + dr3[1] * dr3[1]
238  + dr3[2] * dr3[2];
239  if (rjk < rc2)
240  {
241  // Energy calculation.
242 #ifdef NOCFCACHE
243  double pfcik;
244  double pdfcik;
245  fc.fdf(rik, pfcik, pdfcik);
246 #else
247  double& pfcik = nk.fc;
248  double& pdfcik = nk.dfc;
249  if (nk.cutoffType != cutoffType ||
250  nk.rc != rc ||
251  nk.cutoffAlpha != cutoffAlpha)
252  {
253  fc.fdf(rik, pfcik, pdfcik);
254  nk.rc = rc;
255  nk.cutoffType = cutoffType;
256  nk.cutoffAlpha = cutoffAlpha;
257  }
258 #endif
259  rjk = sqrt(rjk);
260 
261  double pfcjk;
262  double pdfcjk;
263  fc.fdf(rjk, pfcjk, pdfcjk);
264 
265  // SIMPLE EXPRESSIONS:
266  //Vec3D const drik(atom.neighbors[k].dr);
267  //double const* const dr2 = drik.r;
268  //double const* const dr3 = drjk.r;
269  double const rinvijik = 1.0 / rij / rik;
270  // SIMPLE EXPRESSIONS:
271  //double const costijk = (drij * drik) * rinvijik;
272  double const costijk = (dr1[0] * dr2[0] +
273  dr1[1] * dr2[1] +
274  dr1[2] * dr2[2]) * rinvijik;
275  double const z = elementMap.atomicNumber(nej)
276  * elementMap.atomicNumber(nek);
277  double const pfc = z * pfcij * pfcik * pfcjk;
278  double const r2ik = rik * rik;
279  double const pr1 = z * pfcik * pfcjk * pdfcij / rij;
280  double const pr2 = z * pfcij * pfcjk * pdfcik / rik;
281  double const pr3 = z * pfcij * pfcik * pdfcjk / rjk;
282  double vexp = 0.0;
283  double rijs = 0.0;
284  double riks = 0.0;
285  double rjks = 0.0;
286 
287  for (size_t l = 0; l < members.size(); ++l)
288  {
289  if (calculateExp[l])
290  {
291  rijs = rij - rs[l];
292  riks = rik - rs[l];
293  rjks = rjk - rs[l];
294  double const r2sum = rijs * rijs
295  + riks * riks
296  + rjks * rjks;
297  vexp = exp(-eta[l] * r2sum);
298  }
299  double const plambda = 1.0
300  + lambda[l] * costijk;
301  double fg = vexp;
302  if (plambda <= 0.0) fg = 0.0;
303  else
304  {
305  if (useIntegerPow[l])
306  {
307  fg *= pow_int(plambda, zetaInt[l] - 1);
308  }
309  else
310  {
311  fg *= pow(plambda, zeta[l] - 1.0);
312  }
313  }
314  result[l] += fg * plambda * pfc;
315 
316  // Force calculation.
317  if (!derivatives) continue;
318  fg *= factorNorm[l];
319  double const pfczl = pfc * zetaLambda[l];
320  double const p2etapl = plambda * factorDeriv[l];
321  double const p1 = fg * (pfczl * (rinvijik - costijk
322  / r2ij - p2etapl * rijs / rij)
323  + pr1 * plambda);
324  double const p2 = fg * (pfczl * (rinvijik - costijk
325  / r2ik - p2etapl * riks / rik)
326  + pr2 * plambda);
327  double const p3 = fg * (pfczl * (rinvijik + p2etapl
328  * rjks / rjk) - pr3 * plambda);
329 
330  // Save force contributions in Atom storage.
331  //
332  // SIMPLE EXPRESSIONS:
333  //size_t const li = members[l]->getIndex();
334  //atom.dGdr[li] += p1 * drij + p2 * drik;
335  //atom.neighbors[j].dGdr[li] -= p1 * drij
336  // + p3 * drjk;
337  //atom.neighbors[k].dGdr[li] -= p2 * drik
338  // - p3 * drjk;
339 
340  double const p1drijx = p1 * dr1[0];
341  double const p1drijy = p1 * dr1[1];
342  double const p1drijz = p1 * dr1[2];
343 
344  double const p2drikx = p2 * dr2[0];
345  double const p2driky = p2 * dr2[1];
346  double const p2drikz = p2 * dr2[2];
347 
348  double const p3drjkx = p3 * dr3[0];
349  double const p3drjky = p3 * dr3[1];
350  double const p3drjkz = p3 * dr3[2];
351 
352 #ifdef IMPROVED_SFD_MEMORY
353  size_t li = memberIndex[l];
354 #else
355  size_t const li = memberIndex[l];
356 #endif
357  double* dGdr = atom.dGdr[li].r;
358  dGdr[0] += p1drijx + p2drikx;
359  dGdr[1] += p1drijy + p2driky;
360  dGdr[2] += p1drijz + p2drikz;
361 
362 #ifdef IMPROVED_SFD_MEMORY
363  li = memberIndexPerElement[l][nej];
364 #endif
365  dGdr = nj.dGdr[li].r;
366  dGdr[0] -= p1drijx + p3drjkx;
367  dGdr[1] -= p1drijy + p3drjky;
368  dGdr[2] -= p1drijz + p3drjkz;
369 
370 #ifdef IMPROVED_SFD_MEMORY
371  li = memberIndexPerElement[l][nek];
372 #endif
373  dGdr = nk.dGdr[li].r;
374  dGdr[0] -= p2drikx - p3drjkx;
375  dGdr[1] -= p2driky - p3drjky;
376  dGdr[2] -= p2drikz - p3drjkz;
377  } // l
378  } // rjk <= rc
379  } // rik <= rc
380  } // k
381  } // rij <= rc
382  } // j
383 
384  for (size_t l = 0; l < members.size(); ++l)
385  {
386  result[l] *= factorNorm[l] / scalingFactors[l];
387  atom.G[memberIndex[l]] = members[l]->scale(result[l]);
388  }
389 
390  delete[] result;
391 
392  return;
393 }
