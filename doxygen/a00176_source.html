126 void SymmetryFunctionRadial::calculate(Atom& atom,
127  bool const derivatives) const
128 {
129  double result = 0.0;
130 
131  for (size_t j = 0; j < atom.numNeighbors; ++j)
132  {
133  Atom::Neighbor& n = atom.neighbors[j];
134  if (e1 == n.element && n.d < rc)
135  {
136  // Energy calculation.
137  double const rij = n.d;
138  double const pexp = exp(-eta * (rij - rs) * (rij - rs));
139 
140  // Calculate cutoff function and derivative.
141 #ifdef NOCFCACHE
142  double pfc;
143  double pdfc;
144  fc.fdf(rij, pfc, pdfc);
145 #else
146  // If cutoff radius matches with the one in the neighbor storage
147  // we can use the previously calculated value.
148  double& pfc = n.fc;
149  double& pdfc = n.dfc;
150  if (n.cutoffType != cutoffType ||
151  n.rc != rc ||
152  n.cutoffAlpha != cutoffAlpha)
153  {
154  fc.fdf(rij, pfc, pdfc);
155  n.rc = rc;
156  n.cutoffType = cutoffType;
157  n.cutoffAlpha = cutoffAlpha;
158  }
159 #endif
160 
161  result += pexp * pfc;
162  // Force calculation.
163  if (!derivatives) continue;
164  double const p1 = scalingFactor
165  * (pdfc - 2.0 * eta * (rij - rs)
166  * pfc) * pexp / rij;
167  Vec3D dij = p1 * n.dr;
168  // Save force contributions in Atom storage.
169  atom.dGdr[index] += dij;
170 #ifdef IMPROVED_SFD_MEMORY
171  n.dGdr[indexPerElement[e1]] -= dij;
172 #else
173  n.dGdr[index] -= dij;
174 #endif
175  }
176  }
177 
178  atom.G[index] = scale(result);
179 
180  return;
181 }
