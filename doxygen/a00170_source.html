71 bool SymmetryFunctionGroupWeightedRadial::
72 addMember(SymmetryFunction const* const symmetryFunction)
73 {
74  if (symmetryFunction->getType() != type) return false;
75 
76  SymmetryFunctionWeightedRadial const* sf =
77  dynamic_cast<SymmetryFunctionWeightedRadial const*>(symmetryFunction);
78 
79  if (members.empty())
80  {
81  cutoffType = sf->getCutoffType();
82  cutoffAlpha = sf->getCutoffAlpha();
83  ec = sf->getEc();
84  rc = sf->getRc();
85  convLength = sf->getConvLength();
86 
87  fc.setCutoffType(cutoffType);
88  fc.setCutoffRadius(rc);
89  fc.setCutoffParameter(cutoffAlpha);
90  }
91 
92  if (sf->getCutoffType() != cutoffType ) return false;
93  if (sf->getCutoffAlpha() != cutoffAlpha) return false;
94  if (sf->getEc() != ec ) return false;
95  if (sf->getRc() != rc ) return false;
96  if (sf->getConvLength() != convLength )
97  {
98  throw runtime_error("ERROR: Unable to add symmetry function members "
99  "with different conversion factors.\n");
100  }
101 
102  members.push_back(sf);
103 
104  return true;
105 }

135 // Depending on chosen symmetry functions this function may be very
136 // time-critical when predicting new structures (e.g. in MD simulations). Thus,
137 // lots of optimizations were used sacrificing some readablity. Vec3D
138 // operations have been rewritten in simple C array style and the use of
139 // temporary objects has been minmized. Some of the originally coded
140 // expressions are kept in comments marked with "SIMPLE EXPRESSIONS:".
141 void SymmetryFunctionGroupWeightedRadial::calculate(
142  Atom& atom,
143  bool const derivatives) const
144 {
145  double* result = new double[members.size()];
146  for (size_t k = 0; k < members.size(); ++k)
147  {
148  result[k] = 0.0;
149  }
150 
151  for (size_t j = 0; j < atom.numNeighbors; ++j)
152  {
153  Atom::Neighbor& n = atom.neighbors[j];
154  if (n.d < rc)
155  {
156  // Energy calculation.
157  double const rij = n.d;
158  size_t const nej = n.element;
159 
160  // Calculate cutoff function and derivative.
161 #ifdef NOCFCACHE
162  double pfc;
163  double pdfc;
164  fc.fdf(rij, pfc, pdfc);
165 #else
166  // If cutoff radius matches with the one in the neighbor storage
167  // we can use the previously calculated value.
168  double& pfc = n.fc;
169  double& pdfc = n.dfc;
170  if (n.cutoffType != cutoffType ||
171  n.rc != rc ||
172  n.cutoffAlpha != cutoffAlpha)
173  {
174  fc.fdf(rij, pfc, pdfc);
175  n.rc = rc;
176  n.cutoffType = cutoffType;
177  n.cutoffAlpha = cutoffAlpha;
178  }
179 #endif
180  double const* const d1 = n.dr.r;
181  for (size_t k = 0; k < members.size(); ++k)
182  {
183  double pexp = elementMap.atomicNumber(nej)
184  * exp(-eta[k] * (rij - rs[k]) * (rij - rs[k]));
185  result[k] += pexp * pfc;
186  // Force calculation.
187  if (!derivatives) continue;
188  double const p1 = scalingFactors[k] * (pdfc - 2.0 * eta[k]
189  * (rij - rs[k]) * pfc) * pexp / rij;
190  // SIMPLE EXPRESSIONS:
191  //Vec3D const dij = p1 * atom.neighbors[j].dr;
192  double const p1drijx = p1 * d1[0];
193  double const p1drijy = p1 * d1[1];
194  double const p1drijz = p1 * d1[2];
195 
196  // Save force contributions in Atom storage.
197 #ifdef IMPROVED_SFD_MEMORY
198  size_t ki = memberIndex[k];
199 #else
200  size_t const ki = memberIndex[k];
201 #endif
202  // SIMPLE EXPRESSIONS:
203  //atom.dGdr[ki] += dij;
204  //atom.neighbors[j].dGdr[ki] -= dij;
205 
206  double* dGdr = atom.dGdr[ki].r;
207  dGdr[0] += p1drijx;
208  dGdr[1] += p1drijy;
209  dGdr[2] += p1drijz;
210 
211 #ifdef IMPROVED_SFD_MEMORY
212  ki = memberIndexPerElement[k][nej];
213 #endif
214  dGdr = n.dGdr[ki].r;
215  dGdr[0] -= p1drijx;
216  dGdr[1] -= p1drijy;
217  dGdr[2] -= p1drijz;
218  }
219  }
220  }
221 
222  for (size_t k = 0; k < members.size(); ++k)
223  {
224  atom.G[memberIndex[k]] = members[k]->scale(result[k]);
225  }
226 
227  delete[] result;
228 
229  return;
230 }
