102 string SymmetryFunctionGroup::getPrintFormatCommon() const
103 {
104  string s;
105 
106  for (PrintOrder::const_iterator it = printOrder.begin();
107  it != printOrder.end(); ++it)
108  {
109  // If common parameter is present add format string.
110  if (parametersCommon.find(*it) != parametersCommon.end())
111  {
112  s += safeFind(printFormat, (*it)).first + ' ';
113  }
114  // Else if member parameter is present add star marker.
115  else if (parametersMember.find(*it) != parametersMember.end())
116  {
117  string t = safeFind(printFormat, (*it)).second;
118  if (t.size () > 0) t.resize (t.size () - 1);
119  t += '*';
120  s += t + ' ';
121  }
122  // Else add just enough empty spaces.
123  else
124  {
125  s += safeFind(printFormat, (*it)).second + ' ';
126  }
127  }
128  // Remove extra space at the end.
129  if (s.size () > 0) s.resize (s.size () - 1);
130  s += '\n';
131 
132  return s;
133 }
