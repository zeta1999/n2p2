77 bool SymmetryFunctionGroupRadial::
78 addMember(SymmetryFunction const* const symmetryFunction)
79 {
80  if (symmetryFunction->getType() != type) return false;
81 
82  SymmetryFunctionRadial const* sf =
83  dynamic_cast<SymmetryFunctionRadial const*>(symmetryFunction);
84 
85  if (members.empty())
86  {
87  cutoffType = sf->getCutoffType();
88  cutoffAlpha = sf->getCutoffAlpha();
89  ec = sf->getEc();
90  rc = sf->getRc();
91  e1 = sf->getE1();
92  convLength = sf->getConvLength();
93 
94  fc.setCutoffType(cutoffType);
95  fc.setCutoffRadius(rc);
96  fc.setCutoffParameter(cutoffAlpha);
97  }
98 
99  if (sf->getCutoffType() != cutoffType ) return false;
100  if (sf->getCutoffAlpha() != cutoffAlpha) return false;
101  if (sf->getEc() != ec ) return false;
102  if (sf->getRc() != rc ) return false;
103  if (sf->getE1() != e1 ) return false;
104  if (sf->getConvLength() != convLength )
105  {
106  throw runtime_error("ERROR: Unable to add symmetry function members "
107  "with different conversion factors.\n");
108  }
109 
110  members.push_back(sf);
111 
112  return true;
113 }

143 // Depending on chosen symmetry functions this function may be very
144 // time-critical when predicting new structures (e.g. in MD simulations). Thus,
145 // lots of optimizations were used sacrificing some readablity. Vec3D
146 // operations have been rewritten in simple C array style and the use of
147 // temporary objects has been minmized. Some of the originally coded
148 // expressions are kept in comments marked with "SIMPLE EXPRESSIONS:".
149 void SymmetryFunctionGroupRadial::calculate(Atom& atom,
150  bool const derivatives) const
151 {
152  double* result = new double[members.size()];
153  for (size_t k = 0; k < members.size(); ++k)
154  {
155  result[k] = 0.0;
156  }
157 
158  for (size_t j = 0; j < atom.numNeighbors; ++j)
159  {
160  Atom::Neighbor& n = atom.neighbors[j];
161  if (e1 == n.element && n.d < rc)
162  {
163  // Energy calculation.
164  double const rij = n.d;
165 
166  // Calculate cutoff function and derivative.
167 #ifdef NOCFCACHE
168  double pfc;
169  double pdfc;
170  fc.fdf(rij, pfc, pdfc);
171 #else
172  // If cutoff radius matches with the one in the neighbor storage
173  // we can use the previously calculated value.
174  double& pfc = n.fc;
175  double& pdfc = n.dfc;
176  if (n.cutoffType != cutoffType ||
177  n.rc != rc ||
178  n.cutoffAlpha != cutoffAlpha)
179  {
180  fc.fdf(rij, pfc, pdfc);
181  n.rc = rc;
182  n.cutoffType = cutoffType;
183  n.cutoffAlpha = cutoffAlpha;
184  }
185 #endif
186  double const* const d1 = n.dr.r;
187  for (size_t k = 0; k < members.size(); ++k)
188  {
189  double pexp = exp(-eta[k] * (rij - rs[k]) * (rij - rs[k]));
190  result[k] += pexp * pfc;
191  // Force calculation.
192  if (!derivatives) continue;
193  double const p1 = scalingFactors[k] * (pdfc - 2.0 * eta[k]
194  * (rij - rs[k]) * pfc) * pexp / rij;
195  // SIMPLE EXPRESSIONS:
196  //Vec3D const dij = p1 * atom.neighbors[j].dr;
197  double const p1drijx = p1 * d1[0];
198  double const p1drijy = p1 * d1[1];
199  double const p1drijz = p1 * d1[2];
200 
201  // Save force contributions in Atom storage.
202 #ifdef IMPROVED_SFD_MEMORY
203  size_t ki = memberIndex[k];
204 #else
205  size_t const ki = memberIndex[k];
206 #endif
207  // SIMPLE EXPRESSIONS:
208  //atom.dGdr[ki] += dij;
209  //atom.neighbors[j].dGdr[ki] -= dij;
210 
211  double* dGdr = atom.dGdr[ki].r;
212  dGdr[0] += p1drijx;
213  dGdr[1] += p1drijy;
214  dGdr[2] += p1drijz;
215 
216 #ifdef IMPROVED_SFD_MEMORY
217  ki = memberIndexPerElement[k][e1];
218 #endif
219  dGdr = n.dGdr[ki].r;
220  dGdr[0] -= p1drijx;
221  dGdr[1] -= p1drijy;
222  dGdr[2] -= p1drijz;
223  }
224  }
225  }
226 
227  for (size_t k = 0; k < members.size(); ++k)
228  {
229  atom.G[memberIndex[k]] = members[k]->scale(result[k]);
230  }
231 
232  delete[] result;
233 
234  return;
235 }
